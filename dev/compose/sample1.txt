            - ./logs/error.log:/var/log/nginx/error.log
            - ./logs/access.log:/var/log/nginx/access.log

            - vnet

        build: ./phpfpm
        ports:
            - '80:80'

        networks:

        volumes:
            - ./db-data:/var/lib/mysql
        ports:
            - '3306:3306'
        environment:
            MARIADB_ROOT_PASSWORD: root    
        networks:
            - vnet
